% Guard so \nocite{*} does not mark everything as cited
\newif\ifinnocite
\let\oldnocite\nocite
\renewcommand*{\nocite}[1]{%
  \begingroup
  \innocitetrue
  \oldnocite{#1}%
  \endgroup
}

% Mark only genuinely cited items
\AtEveryCitekey{%
  \ifinnocite
  \else
    \addtocategory{cited}{\thefield{entrykey}}%
  \fi
}

% Color uncited entries
\AtEveryBibitem{%
  \ifcategory{cited}
    {}
    {\color{gray}}%
}
